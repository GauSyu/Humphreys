       #3 & \longmapsto #4
    \end{align*}
  }

%                        Typical Commutative Diagrams
%
\newcommand{\initial}[6]
{
\begin{displaymath}
   \xymatrix{
     {#1} \ar[r]^{#2} \ar[dr]_{#4} & {#3} \ar@{-->}[d]^{#6} \\
     & {#5}
   }
\end{displaymath}
}
\newcommand{\terminal}[6]
{
\begin{displaymath}
   \xymatrix{
     {#5} \ar[dr]^{#4} \ar@{-->}[d]_{#6} & \\
     {#3} \ar[r]_{#2} & {#1}
   }
\end{displaymath}
}
\newcommand{\functor}[8]
{
\begin{displaymath}
   \xymatrix{
     {#1}\ar[d]_{#2}\ar[r]^{#4} & {#5}\ar[d]^{#8} \\
     {#3}\ar[r]_{#6} & {#7}
   }
\end{displaymath}
}

\newcommand{\Functor}[9]
{
\begin{displaymath}
   \xymatrix{
     {#1}\ar[rr]^{#2} & & {#3}\\
     {#4}\ar[dd]^{#5}="a" \ar@{|->}[rr]  & & {#7}\ar[dd]_{#8}="b" \ar@{|->} "a";"b"\\
     & & \\
     {#6}\ar@{|->}[rr] & & {#9}
   }
\end{displaymath}
}
